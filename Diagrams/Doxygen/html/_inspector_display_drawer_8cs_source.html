146 object GetValueRecursive(object obj, int index, string[] paths)
147 {
148 var path = paths[index];
149
150 FieldInfo fldInfo = null;
151 var type = obj.GetType();
152 while (fldInfo == null)
153 {
154 // attempt to get information about the field
155 fldInfo = type.GetField(path, BindingFlags.IgnoreCase | BindingFlags.GetField | BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic);
156
157 if (fldInfo != null ||
158 type.BaseType == null ||
159 type.BaseType.IsSubclassOf(typeof(ReactiveProperty<>))) break;
160
161 // if the field information is missing, it may be in the base class
162 type = type.BaseType;
163 }
164
165 // If array, path = Array.data[index]
166 if (fldInfo == null && path == "Array")
167 {
168 try
169 {
170 path = paths[++index];
171 var m = Regex.Match(path, @"(.+)\[([0-9]+)*\]");
172 var arrayIndex = int.Parse(m.Groups[2].Value);
173 var arrayValue = (obj as System.Collections.IList)[arrayIndex];
174 if (index < paths.Length - 1)
175 {
176 return GetValueRecursive(arrayValue, ++index, paths);
177 }
178 else
179 {
180 return arrayValue;
181 }
182 }
183 catch
184 {
185 Debug.Log("InspectorDisplayDrawer Exception, objType:" + obj.GetType().Name + " path:" + string.Join(", ", paths));
186 throw;
187 }
188 }
189 else if (fldInfo == null)
190 {
191 throw new Exception("Can't decode path, please report to UniRx's GitHub issues:" + string.Join(", ", paths));
192 }
193
194 var v = fldInfo.GetValue(obj);
195 if (index < paths.Length - 1)
196 {
197 return GetValueRecursive(v, ++index, paths);
198 }
199
200 return v;
