7using UnityEngine;

9namespace UniRx.Examples

11 // sample script, attach your object.
12 public class Sample01_ObservableWWW : MonoBehaviour
13 {
14 void Start()
15 {
16 // Basic: Download from google.
17 {
18 ObservableWWW.Get("http://google.co.jp/")
19 .Subscribe(
20 x => Debug.Log(x.Substring(0, 100)), // onSuccess
21 ex => Debug.LogException(ex)); // onError
22 }
23
24 // Linear Pattern with LINQ Query Expressions
25 // download after google, start bing download
26 {
27 var query = from google in ObservableWWW.Get("http://google.com/")
28 from bing in ObservableWWW.Get("http://bing.com/")
29 select new { google, bing };
30
31 var cancel = query.Subscribe(x => Debug.Log(x.google.Substring(0, 100) + ":" + x.bing.Substring(0, 100)));
32
33 // Call Dispose is cancel downloading.
34 cancel.Dispose();
35 }
36
37 // Observable.WhenAll is for parallel asynchronous operation
38 // (It's like Observable.Zip but specialized for single async operations like Task.WhenAll of .NET 4)
39 {
40 var parallel = Observable.WhenAll(
41 ObservableWWW.Get("http://google.com/"),
42 ObservableWWW.Get("http://bing.com/"),
43 ObservableWWW.Get("http://unity3d.com/"));
44
45 parallel.Subscribe(xs =>
46 {
47 Debug.Log(xs[0].Substring(0, 100)); // google
48 Debug.Log(xs[1].Substring(0, 100)); // bing
49 Debug.Log(xs[2].Substring(0, 100)); // unity
50 });
51 }
52
53 // with Progress
54 {
55 // notifier for progress
56 var progressNotifier = new ScheduledNotifier<float>();
57 progressNotifier.Subscribe(x => Debug.Log(x)); // write www.progress
58
59 // pass notifier to WWW.Get/Post
60 ObservableWWW.Get("http://google.com/", progress: progressNotifier).Subscribe();
61 }
62
63 // with Error
64 {
65 // If WWW has .error, ObservableWWW throws WWWErrorException to onError pipeline.
66 // WWWErrorException has RawErrorMessage, HasResponse, StatusCode, ResponseHeaders
67 ObservableWWW.Get("http://www.google.com/404")
68 .CatchIgnore((WWWErrorException ex) =>
69 {
70 Debug.Log(ex.RawErrorMessage);
71 if (ex.HasResponse)
72 {
73 Debug.Log(ex.StatusCode);
74 }
75 foreach (var item in ex.ResponseHeaders)
76 {
77 Debug.Log(item.Key + ":" + item.Value);
78 }
79 })
80 .Subscribe();
81 }
82 }
83 }
