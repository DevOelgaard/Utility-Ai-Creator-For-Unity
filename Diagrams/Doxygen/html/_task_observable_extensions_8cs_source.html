49 private static IObservable<Unit> ToObservableImpl(Task task, IScheduler scheduler)
50 {
51 var res = default(IObservable<Unit>);
52
53 if (task.IsCompleted)
54 {
55 scheduler = scheduler ?? Scheduler.Immediate;
56
57 switch (task.Status)
58 {
59 case TaskStatus.RanToCompletion:
60 res = Observable.Return<Unit>(Unit.Default, scheduler);
61 break;
62 case TaskStatus.Faulted:

64 break;
65 case TaskStatus.Canceled:
66 res = Observable.Throw<Unit>(new TaskCanceledException(task), scheduler);
67 break;
68 }
69 }
70 else
71 {
72 //
73 // Separate method to avoid closure in synchronous completion case.
74 //
75 res = ToObservableSlow(task, scheduler);

204 private static TaskContinuationOptions GetTaskContinuationOptions(IScheduler scheduler)
205 {
206 var options = TaskContinuationOptions.None;
207
208 if (scheduler != null)
209 {
210 //
211 // We explicitly don't special-case the immediate scheduler here. If the user asks for a
212 // synchronous completion, we'll try our best. However, there's no guarantee due to the
213 // internal stack probing in the TPL, which may cause asynchronous completion on a thread
214 // pool thread in order to avoid stack overflows. Therefore we can only attempt to be more
215 // efficient in the case where the user specified a scheduler, hence we know that the
216 // continuation will trigger a scheduling operation. In case of the immediate scheduler,
217 // it really becomes "immediate scheduling" wherever the TPL decided to run the continuation,
218 // i.e. not necessarily where the task was completed from.
219 //
220 options |= TaskContinuationOptions.ExecuteSynchronously;
221 }
222
223 return options;
224 }

294 public static Task<TResult> ToTask<TResult>(this IObservable<TResult> observable, CancellationToken cancellationToken, object state)
295 {
296 if (observable == null)
297 throw new ArgumentNullException("observable");
298
299 var hasValue = false;
300 var lastValue = default(TResult);
301
302 var tcs = new TaskCompletionSource<TResult>(state);
303
304 var disposable = new SingleAssignmentDisposable();
305
306 var ctr = default(CancellationTokenRegistration);
307
308 if (cancellationToken.CanBeCanceled)
309 {
310 ctr = cancellationToken.Register(() =>
311 {
312 disposable.Dispose();
313 tcs.TrySetCanceled(cancellationToken);
314 });
315 }
316
317 var taskCompletionObserver = Observer.Create<TResult>(
318 value =>
319 {
320 hasValue = true;
321 lastValue = value;
322 },
323 ex =>
324 {
325 tcs.TrySetException(ex);
326
327 ctr.Dispose(); // no null-check needed (struct)
328 disposable.Dispose();
329 },
330 () =>
331 {
332 if (hasValue)
333 tcs.TrySetResult(lastValue);
334 else
335 tcs.TrySetException(new InvalidOperationException("Strings_Linq.NO_ELEMENTS"));
336
337 ctr.Dispose(); // no null-check needed (struct)
338 disposable.Dispose();
339 }
340 );
341
342 //
343 // Subtle race condition: if the source completes before we reach the line below, the SingleAssigmentDisposable
344 // will already have been disposed. Upon assignment, the disposable resource being set will be disposed on the
345 // spot, which may throw an exception. (Similar to TFS 487142)
346 //
347 try
348 {
349 //
350 // [OK] Use of unsafe Subscribe: we're catching the exception here to set the TaskCompletionSource.
351 //
352 // Notice we could use a safe subscription to route errors through OnError, but we still need the
353 // exception handling logic here for the reason explained above. We cannot afford to throw here
354 // and as a result never set the TaskCompletionSource, so we tunnel everything through here.
355 //
356 disposable.Disposable = observable.Subscribe/*Unsafe*/(taskCompletionObserver);
357 }
358 catch (Exception ex)
359 {
360 tcs.TrySetException(ex);
361 }
362
363 return tcs.Task;
364 }
