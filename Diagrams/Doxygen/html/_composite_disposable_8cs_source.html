120 //
121 // List<T> doesn't shrink the size of the underlying array but does collapse the array
122 // by copying the tail one position to the left of the removal index. We don't need
123 // index-based lookup but only ordering for sequential disposal. So, instead of spending
124 // cycles on the Array.Copy imposed by Remove, we use a null sentinel value. We also
125 // do manual Swiss cheese detection to shrink the list if there's a lot of holes in it.
126 //
127 var i = _disposables.IndexOf(item);
