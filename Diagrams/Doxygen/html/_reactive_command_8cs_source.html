221 public IDisposable Execute(T parameter)
222 {
223 if (canExecute.Value)
224 {
225 canExecuteSource.Value = false;
226 var a = asyncActions.Data;
227 if (a.Length == 1)
228 {
229 try
230 {
231 var asyncState = a[0].Invoke(parameter) ?? Observable.ReturnUnit();
232 return asyncState.Finally(() => canExecuteSource.Value = true).Subscribe();
233 }
234 catch
235 {
236 canExecuteSource.Value = true;
237 throw;
238 }
239 }
240 else
241 {
242 var xs = new IObservable<Unit>[a.Length];
243 try
244 {
245 for (int i = 0; i < a.Length; i++)
246 {
247 xs[i] = a[i].Invoke(parameter) ?? Observable.ReturnUnit();
248 }
249 }
250 catch
251 {
252 canExecuteSource.Value = true;
253 throw;
254 }
255
256 return Observable.WhenAll(xs).Finally(() => canExecuteSource.Value = true).Subscribe();
257 }
258 }
259 else
260 {
261 return Disposable.Empty;
262 }
263 }
